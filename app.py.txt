import streamlit as st
import pandas as pd
from streamlit_drawable_canvas import st_canvas
from PIL import Image

st.set_page_config(page_title="مكتب الحسنين - جرد الحساسات", layout="wide")

st.title("نظام جرد الحساسات التفاعلي - الطابق الأول")
st.write("انقر على موقع الحساس في المخطط لتسجيل رقمه (واقع الحال)")

# 1. تحميل صورة المخطط
# ملاحظة: قم بتسمية صورة المخطط floor_plan.png ووضعها في نفس المجلد
bg_image = Image.open("floor_plan.png")
width, height = bg_image.size

# 2. إعدادات لوحة الرسم التفاعلية
canvas_result = st_canvas(
    fill_color="rgba(255, 165, 0, 0.3)",  # لون الدائرة عند النقر
    stroke_width=2,
    background_image=bg_image,
    update_streamlit=True,
    height=height,
    width=width,
    drawing_mode="point", # وضع النقر لتحديد النقاط
    key="canvas",
)

# 3. معالجة البيانات المسجلة
if canvas_result.json_data is not None:
    objects = pd.json_normalize(canvas_result.json_data["objects"])
    if not objects.empty:
        st.subheader("قائمة الحساسات المسجلة")
        
        # إضافة عمود لإدخال رقم الحساس لكل نقطة تم النقر عليها
        objects['sensor_id'] = ""
        for index, row in objects.iterrows():
            objects.at[index, 'sensor_id'] = st.text_input(
                f"أدخل رقم الحساس للنقطة {index+1}:", 
                key=f"input_{index}"
            )
        
        # عرض الجدول
        df_final = objects[['left', 'top', 'sensor_id']]
        st.table(df_final)

        # 4. تصدير البيانات إلى إكسل
        if st.button("تصدير جدول الحساسات إلى Excel"):
            df_final.to_excel("sensor_audit.xlsx", index=False)
            st.success("تم حفظ الملف باسم sensor_audit.xlsx")
